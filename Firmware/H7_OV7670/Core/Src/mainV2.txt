/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MPU Configuration--------------------------------------------------------*/
  MPU_Config();

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_USART3_UART_Init();
  MX_I2C2_Init();
  MX_USB_DEVICE_Init();
  MX_DCMI_Init();
  /* USER CODE BEGIN 2 */
  // Don't exactly know why but these caches need to be manually enabled
  SCB_EnableICache();
  SCB_EnableDCache();

  // us timer initialize
  DWT_Init();

  // Camera initialize TODO: significant changes needed when moving to epc660
  HAL_GPIO_WritePin(GPIOD, GPIO_PIN_12, GPIO_PIN_RESET);
  ov7670_init(&hdcmi, &hdma_dcmi, &hi2c2);
  ov7670_config(OV7670_MODE_QVGA_RGB565);
  ov7670_stopCap();

  // I don't know if the frame header is useful - can be removed later
  const uint8_t FRAME_HEADER[4] = {0xAA, 0x55, 0xAA, 0x55};

  // ms precision timers
  uint32_t last_frame_tick = 0;
  uint32_t loop_tick = 0;
  uint32_t frame_timeout_ms = 1000;

  float cpu_freq_mhz = (float)HAL_RCC_GetHCLKFreq() / 1000000.0f;

  // Clear flags (These are triggered in callbacks)
  all_frames_ready = 0;
  frame_capture_count = 0;

  // Start the DCMI -> waits for first VSYNC
  uint32_t dma_length = (SINGLE_FRAME_SIZE * NUM_FRAMES) / 2;
  HAL_DCMI_Start_DMA(&hdcmi, DCMI_MODE_CONTINUOUS, (uint32_t)quad_frame_buffer, dma_length);

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {

	  // Blinking light to make sure main loop is still running
	  if (HAL_GetTick() - loop_tick > frame_timeout_ms)
	  {
		  HAL_GPIO_TogglePin(GPIOE, GPIO_PIN_1);
		  loop_tick = HAL_GetTick();
	  }

	  // all_frames_ready set by DCMI frame event callback
	  if(all_frames_ready == 1)
	  {

		// End timer for frame capture (happens inside DCMI frame even callback)
		my_printf("DCMI frame capture duration: %.3f\r\n", (float)(tDCMI_end - tDCMI_start) / cpu_freq_mhz);
	    all_frames_ready = 0;


	    // **************************** DEPTH CALCULATION START ***********************************
	    tDepthCalc_start = DWT->CYCCNT;
	    SCB_InvalidateDCache_by_Addr((uint32_t*)quad_frame_buffer, SINGLE_FRAME_SIZE * NUM_FRAMES * 2);
	    simulate_tof_data(simulate_cube_visible);
	    calculate_depth_simple();
	    SCB_CleanDCache_by_Addr((uint32_t*)quad_frame_buffer, SINGLE_FRAME_SIZE * NUM_FRAMES * 2);

	    // Sets current frame to calibration frame if button is pressed
	    if (calibration_requested == 1)
	    {
	      simulate_cube_visible = !(simulate_cube_visible);
	      simulate_tof_data(simulate_cube_visible);
	      calculate_depth_simple();
	      // Copy the frame just calculated into the reference buffer.
	      memcpy(reference_angle_buffer, current_angle_buffer, SINGLE_FRAME_SIZE * 2);

	      // Clean cache
	      SCB_CleanDCache_by_Addr((uint32_t*)reference_angle_buffer, SINGLE_FRAME_SIZE * 2);

	      // Clear the calibration flag
	      calibration_requested = 0;
	      my_printf("Reference frame CALIBRATED.\r\n");
	    }

	    // Check displaced pixels
	    uint32_t displaced_pixels = detect_displacement();
	    my_printf("Displaced pixels: %d\r\n", displaced_pixels);

	    // End timer for depth calculation
	    tDepthCalc_end = DWT->CYCCNT;
		my_printf("Depth calculation duration: %.3f\r\n", (float)(tDepthCalc_end - tDepthCalc_start) / cpu_freq_mhz);
	    // **************************** DEPTH CALCULATION END ***********************************


		// **************************** START DCMI IN BACKGROUND START***********************************
		tDCMI_start = DWT->CYCCNT;
		HAL_DCMI_Start_DMA(&hdcmi, DCMI_MODE_CONTINUOUS, (uint32_t)quad_frame_buffer, dma_length);
		// **************************** START DCMI IN BACKGROUND END ***********************************


	    // **************************** TRANSMIT THROUGH USB START ***********************************
	    tUSB_start = DWT->CYCCNT;
	    while(hUsbDeviceHS.dev_state != USBD_STATE_CONFIGURED);	// Waiting loop

	    USB_Transmit_Blocking((uint8_t*)FRAME_HEADER, 4);

	    uint8_t* p_buffer = (uint8_t*)current_angle_buffer;
	    uint32_t bytes_remaining = 320 * 240 * 2; // 153,600
	    uint16_t chunk_size;

	    while (bytes_remaining > 0)
	    {
	      // Calculate the size of the next chunk
	      if (bytes_remaining > 65535) {
	          chunk_size = 65535;
	      } else {
	          chunk_size = (uint16_t)bytes_remaining;
	      }

	      // Send the chunk
	      USB_Transmit_Blocking(p_buffer, chunk_size);

	      // Move the pointer and update the remaining count
	      p_buffer += chunk_size;
	      bytes_remaining -= chunk_size;
	    }

	    // End timer for usb transmit
	    tUSB_end = DWT->CYCCNT;
	    my_printf("USB frame transmission: %.3f\r\n", (float)(tUSB_end - tUSB_start) / cpu_freq_mhz);
	    // **************************** TRANSMIT THROUGH USB END ***********************************

	    // Check if
	    last_frame_tick = HAL_GetTick();
	  }

	  // Reinitialize if DCMI gets stuck
	  if (HAL_GetTick() - last_frame_tick > frame_timeout_ms)
	  {
		HAL_DCMI_Stop(&hdcmi);
		HAL_DCMI_DeInit(&hdcmi);
		MX_DCMI_Init(); // call your CubeMX-generated init function again

		HAL_DCMI_Start_DMA(&hdcmi, DCMI_MODE_CONTINUOUS, (uint32_t)quad_frame_buffer, dma_length);
		last_frame_tick = HAL_GetTick();
	  }


    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}

void HAL_DCMI_FrameEventCallback(DCMI_HandleTypeDef *hdcmi)
{
  frame_capture_count++;

  if (frame_capture_count >= NUM_FRAMES)
  {
	  HAL_DCMI_Stop(hdcmi);
	  all_frames_ready = 1;
	  frame_capture_count = 0;

	  // Stop frame capture timer
	  tDCMI_end = DWT->CYCCNT;
  }
}

void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
{
  // Check if the interrupt came from our button (PC13)
  if (GPIO_Pin == GPIO_PIN_13)
  {
    // Set the flag for the main loop to handle.
    // This keeps the interrupt routine extremely fast and safe.
    calibration_requested = 1;
    my_printf("INTERRUPTED!");
  }
}

void vprint(const char *fmt, va_list argp) {
	char string[200];
	if (0 < vsprintf(string, fmt, argp)) // build string
	{
		HAL_UART_Transmit(&huart3, (uint8_t*) string, strlen(string), 0xffffff); // send message via UART
	}
}

void my_printf(const char *fmt, ...) // custom printf() function
{
	va_list argp;
	va_start(argp, fmt);
	vprint(fmt, argp);
	va_end(argp);
}

	}
}

    }
}

void DWT_Init(void)
{
  if (!(CoreDebug->DEMCR & CoreDebug_DEMCR_TRCENA_Msk))
  {
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
  }
  if (!(DWT->CTRL & DWT_CTRL_CYCCNTENA_Msk))
  {
    DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
  }
  DWT->CYCCNT = 0;
}

#define ERROR_THRESHOLD 1820

void calculate_depth_simple(void)
{
    for (int i = 0; i < (320 * 240); i++)
    {
        // 1. Read, cast, and subtract
        float y = (float)p_frame1[i] - (float)p_frame3[i];
        float x = (float)p_frame2[i] - (float)p_frame4[i];

        // 2. Calculate and store
        float phase = atan2f(y, x);

        uint16_t scaled_angle = (uint16_t)((phase + M_PI) * PHASE_TO_U16_SCALE);

        current_angle_buffer[i] = scaled_angle;
    }


}

uint32_t detect_displacement(void)
{
	uint32_t displaced_pixel_count = 0;

	for (int i = 0; i < SINGLE_FRAME_SIZE; i++)
	{
		int32_t diff = abs((int32_t)current_angle_buffer[i] - (int32_t)reference_angle_buffer[i]);
		if (diff > 32768)
		{
			diff = 65536 - diff;
		}

		if (diff > ERROR_THRESHOLD)
		{
			displaced_pixel_count++;
		}
	}
	return displaced_pixel_count;
}

void simulate_tof_data(uint8_t show_cube)
{
  float phase_to_use;
  uint16_t amp = (uint16_t)SIM_AMPLITUDE;
  uint16_t offset = (uint16_t)SIM_OFFSET;

  const int door_center_x = IMG_WIDTH / 2;
  const int door_half_w = DOOR_WIDTH / 2;
  const int door_bottom = DOOR_BOTTOM;
  const int door_top = door_bottom - DOOR_HEIGHT;
  const int door_center_y = (door_top + door_bottom) / 2;
  const int floor_start = door_bottom;

  static int intruder_x = -INTRUDER_SIZE;
  static int intruder_dir = 1;
  const int intruder_y = door_center_y - (INTRUDER_SIZE / 2);
  static uint32_t noise_state = 0x13579BDFu;

  if (!show_cube)
  {
    intruder_x = door_center_x - door_half_w - INTRUDER_SIZE;
    intruder_dir = 1;
  }
  else
  {
    intruder_x += intruder_dir * INTRUDER_SPEED;
    int max_x = door_center_x + door_half_w + INTRUDER_SIZE;
    int min_x = door_center_x - door_half_w - INTRUDER_SIZE;

    if (intruder_x > max_x)
    {
      intruder_x = min_x;
      intruder_dir = 1;
    }
  }

  // Loop over every pixel
  for (int y = 0; y < IMG_HEIGHT; y++)
  {
    float vertical_ratio = (float)y / (float)(IMG_HEIGHT - 1);
    for (int x = 0; x < IMG_WIDTH; x++)
    {
      // Start with wall plane depth and add slight variation to keep things organic
      phase_to_use = PHASE_WALL_NEAR + (vertical_ratio * WALL_PHASE_VARIATION);
      phase_to_use += 0.025f * sinf((float)x * 0.05f);

      // Ramp the floor depth from camera (bottom) up to the wall seam
      if (y >= floor_start)
      {
        float floor_ratio = (float)(y - floor_start) / (float)(IMG_HEIGHT - floor_start);
        phase_to_use = FLOOR_NEAR_PHASE + floor_ratio * (PHASE_WALL_NEAR - FLOOR_NEAR_PHASE);
      }

      int dx = x - door_center_x;
      int inside_door = (abs(dx) <= door_half_w) && (y >= door_top) && (y <= door_bottom);
      int on_frame = (abs(abs(dx) - door_half_w) < DOOR_FRAME_THICKNESS && y >= door_top && y <= door_bottom) ||
               (y >= door_top && y < door_top + DOOR_FRAME_THICKNESS && abs(dx) <= door_half_w);

      if (inside_door)
      {
        float horizontal = (float)abs(dx) / (float)door_half_w;
        float vertical = (float)abs(y - door_center_y) / (float)(DOOR_HEIGHT / 2);
        float portal_ratio = fminf(1.0f, sqrtf(horizontal * horizontal + vertical * vertical));
        phase_to_use = PHASE_WALL_NEAR + WALL_PHASE_VARIATION + portal_ratio * ROOM2_PHASE_EXTRA;
      }
      else if (on_frame)
      {
        phase_to_use = DOOR_FRAME_PHASE;
      }

      int intruder_in_row = (y >= intruder_y) && (y < (intruder_y + INTRUDER_SIZE));
      int intruder_in_col = (x >= intruder_x) && (x < (intruder_x + INTRUDER_SIZE));
      int intruder_in_aperture = intruder_in_row && intruder_in_col && inside_door;
      if (show_cube && intruder_in_aperture)
      {
        phase_to_use = INTRUDER_PHASE;
      }

      // Inject lightweight pseudo-random noise to mimic sensor variance
      noise_state = noise_state * 1664525u + 1013904223u;
      int16_t noise_raw = (int16_t)(noise_state >> 16);
      float noise = ((float)noise_raw / 32768.0f) * SIM_NOISE_PHASE;
      phase_to_use += noise;

      // --- 2. Calculate the four frame values for this pixel ---
      float sinP = sinf(phase_to_use);
      float cosP = cosf(phase_to_use);
      int idx = (y * IMG_WIDTH) + x;

      p_frame1[idx] = offset + (uint16_t)(amp * sinP);
      p_frame2[idx] = offset + (uint16_t)(amp * cosP);
      p_frame3[idx] = offset - (uint16_t)(amp * sinP);
      p_frame4[idx] = offset - (uint16_t)(amp * cosP);
    }
  }
}
